#include<stdio.h>

struct node
{
    unsigned dist[20];
    unsigned from[20];
}rt[10];


int main()
{
    int dist[20][20];
    int n,i,j,k,count;
    
    printf("Enter the number of nodes\n");
    scanf("%d",&n);
    
    printf("Enter the distace vector matrix\n");
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            scanf("%d",dist[i][j]);
            dist[i][i] = 0;
            rt[i].dist[j] = dist[i][j];
            rt[i].from[j] = j;
        }
    }
    
    do
    {
        count = 0;
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
                for(k=0;k<n;k++)
                {
                    if(rt[i].dist[j] > dist[i][k] + rt[k].dist[j])
                    {
                        rt[i].dist[j] = rt[i].dist[k]+rt[k].dist[j];
                        rt[i].from[j] = k;
                        count++;
                    }
                }
            }
        }
    }while(count!=0);
    
    for(i=0;i<n;i++)
    {
        printf("State value for %d : \n", i+1);
        printf("\nNode\tVia\tDist.");
        for(j=0;j<n;j++)
        {
            printf("\n%d \t %d \t %d",j+1,rt[i].from[j]+1, rt[i].dist[j]);
        }
    }
    printf("\n\n");
}
